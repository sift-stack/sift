 54class TimeColumn(ConfigBaseModel):
 55    """
 56    Defines a time column entry in the CSV config.
 57    """
 58
 59    format: Union[str, TimeFormatType]
 60    column_number: int
 61    relative_start_time: Optional[str] = None
 62
 63    @field_validator("format", mode="before")
 64    @classmethod
 65    def convert_format(cls, raw: Union[str, TimeFormatType]) -> str:
 66        """
 67        Converts the provided format value to a string.
 68        """
 69        if isinstance(raw, TimeFormatType):
 70            return raw.as_human_str()
 71        elif isinstance(raw, str):
 72            value = TimeFormatType.from_str(raw)
 73            if value is not None:
 74                return value.as_human_str()
 75
 76        raise PydanticCustomError("invalid_config_error", f"Invalid time format: {raw}.")
 77
 78    @model_validator(mode="after")
 79    def validate_time(self) -> Self:
 80        """
 81        Validates the provided time format.
 82        """
 83        format = TimeFormatType.from_str(self.format)  # type: ignore
 84        if format is None:
 85            raise PydanticCustomError(
 86                "invalid_config_error", f"Invalid time format: {self.format}."
 87            )
 88
 89        if format.is_relative():
 90            if self.relative_start_time is None:
 91                raise PydanticCustomError("invalid_config_error", "Missing 'relative_start_time'")
 92        else:
 93            if self.relative_start_time is not None:
 94                raise PydanticCustomError(
 95                    "invalid_config_error",
 96                    "'relative_start_time' specified for non relative time format.",
 97                )
 98
 99        return self

119        """
120        Converts the provided data_type value to a string.
121        """
122        if isinstance(raw, type):
123            if raw == int:
124                return ChannelDataType.INT_64.as_human_str(api_format=True)
125            elif raw == float:
126                return ChannelDataType.DOUBLE.as_human_str(api_format=True)
127            elif raw == str:
128                return ChannelDataType.STRING.as_human_str(api_format=True)
129            elif raw == bool:
130                return ChannelDataType.BOOL.as_human_str(api_format=True)
131        elif isinstance(raw, ChannelDataType):
132            return raw.as_human_str(api_format=True)
133        elif isinstance(raw, str):
134            value = ChannelDataType.from_str(raw)
135            if value is not None:
136                return value.as_human_str(api_format=True)
137
138        raise PydanticCustomError("invalid_config_error", f"Invalid data_type: {raw}.")
139
140    @model_validator(mode="before")
141    @classmethod
142    def concatenate_component_and_name(cls, data: Any) -> Any:
143        """
144        Concatenates Component and Name. If Component is not an empty string, raises a deprecation warning.
145        """
146        if isinstance(data, dict):
147            if "component" in data.keys() and "name" in data.keys():
148                _component_deprecation_warning()
149                data["name"] = channel_fqn(name=data["name"], component=data["component"])
150                data.pop("component")
151        return data
